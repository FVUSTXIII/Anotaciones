* DADO UN NUMERO CLASIFICAR DE LA EDAD DE ACUERDO A LOS SIGUIENTES RANGOS: (BEBE(0-2), NIÑO(3-10), ADOLESCENTE(11-17), ADULTO(18-65), PERSONA MAYOR(66-110), MUERTO(>111))

INICIO
 INT EDAD;
 MOSTRAR "INGRESE SU EDAD: ";
 LEER EDAD;
 IF(EDAD <= 2 && EDAD >= 0) {
  -INICIO
  MOSTRAR "ERES UN BEBÉ";
  -FIN
 }
 IF ELSE(EDAD <= 10 && EDAD >= 3) {
  -INICIO
  MOSTRAR "ERES UN NIÑO";
  -FIN
 }
 IF ELSE(EDAD <= 17 && EDAD >= 11) {
  -INICIO
  MOSTRAR "ERES UN ADOLESCENTE"; 
  -FIN
 }
 IF ELSE(EDAD <= 18 && EDAD >= 65) {
  -INICIO
  MOSTRAR "ERES UN ADULTO";
  -FIN
 }
 IF ELSE(EDAD <= 66 && EDAD >= 110) {
  -INICIO
  MOSTRAR "ERES UNA PERSONA MAYOR";
  -FIN
 } 
 IF ELSE(EDAD > 111) {
  -INICIO
  MOSTRAR "ESTAS MUERTO";
  -FIN
 }
FIN

* FACTORIAL

 INICIO
 INT DIGITO = 0, FACTORIAL = 0, AUX = 0;
 MOSTRAR "INGRESE EL NÚMERO: ";
 LEER DIGITO;
 IF(DIGITO<0) {
  -INICIO
   AUX = DIGITO;
   FACTORIAL = DIGITO;
   FOR(INT I=1; I<AUX; I++) {
   -INICIO 
    DIGITO--;
    FACTORIAL *= DIGITO;
   -FIN
   }
   MOSTRAR "EL FACTORIAL DEL NUMERO INGRESADO ES: " + FACTORIAL;
  -FIN
 }
 FIN
